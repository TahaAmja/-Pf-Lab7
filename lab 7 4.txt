#include<stdio.h>
int main()
{
int str[ ];
printf("Enter digits");
scanf("%d",str);
int n=sizeof(str)/sizeof(str[0]);
int min=str[0];
int max=str[0];
for(int i=0;i<=n;i++)
{
if(str[i]<min)
min=str[i];
if(str[i]>max)
max=str[i];
}
printf("Max=%d",max);
printf("Min=%d",min);
return 0;


}
